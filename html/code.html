아두이노 코드
#include

SoftwareSerial BTSerial(2, 3);

void setup() {
  Serial.begin(9600);
  BTSerial.begin(9600);
  pinMode(11, OUTPUT);
}

void loop() {
  if (BTSerial.available()) {
    char data = BTSerial.read();
    Serial.write(data);

    if(data == 'n'){
      digitalWrite(11, HIGH);
    } else if(data == 'f'){
      digitalWrite(11, LOW);
    }
  }

  if (Serial.available())
    BTSerial.write(Serial.read());
}
